function doPost(e) { try { // Get active sheet var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); // Log received data for debugging Logger.log('Received: ' + JSON.stringify(e)); // Extract data from FormData (e.parameter) var data = e.parameter; // Append row sheet.appendRow([ data.Timestamp || new Date(), data.Name || '', data.Phone || '', data.Age || '', data.Gender || '', data.Profession || '', data.Source || 'Website' ]); return ContentService.createTextOutput("Success"); } catch (error) { Logger.log('Error: ' + error); return ContentService.createTextOutput("Error: " + error); } }
